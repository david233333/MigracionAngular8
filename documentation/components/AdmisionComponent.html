import {Component, OnInit, Output, EventEmitter } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

import { AdmisionViewModel } from './admision.view-model';
import {Admision} from '../../../domain/model/remision/entity/admision.model';
import * as moment from 'moment';
import {Respuesta} from '../../../infraestructure/comun/models/respuesta.model';
import {AdmisionService} from '../../../domain/usecase/remision/admision.service';
import {DatosPacienteService} from '../../../domain/usecase/remision/datos-paciente.service';
import {MensajesService} from '../../../shared/services/mensajes.service';
import {validate} from 'codelyzer/walkerFactory/walkerFn';
import {forEach} from '@angular/router/src/utils/collection';
import {promise} from 'selenium-webdriver';
import consume = promise.consume;
import { ModalComponent } from './modal/modal.component';
import {MatDialog} from '@angular/material';

@Component({
  selector: 'sura-admision',
  templateUrl: './admision.component.html',
  styleUrls: ['./admision.component.scss']
})
export class AdmisionComponent implements OnInit {
  @Output()
  private continuar: EventEmitter<boolean> = new EventEmitter<boolean>();

  @Output()
  private cargando: EventEmitter<boolean> = new EventEmitter<boolean>();

  private formulario: FormGroup;

  public admisionViewModel: AdmisionViewModel = this.iniciarViewModel();

  foods = [
    {value: 'steak-0', viewValue: 'Steak'},
    {value: 'pizza-1', viewValue: 'Pizza'},
    {value: 'tacos-2', viewValue: 'Tacos'}
  ];

  constructor(private fb: FormBuilder,
              private admisionService: AdmisionService,
              private mensajesService: MensajesService,
              private dialog: MatDialog) {
    this.crearFormulario();
    this.suscribirFormulario();
  }

  ngOnInit() {
    this.getDatos();
  }

  /**
   * Obtiene los datos Maestros
   */
  private getDatos(): void {
    this.getPisos();
    this.getProgramas();
  }

  /**
   * Suscribe los cambios del formulario                                                                                                                                                                                 rio
   */
  private suscribirFormulario(): void {
    this.formulario.valueChanges.subscribe(
      formulario => {
        this.continuar.emit(this.formulario.valid);
      }
    );
  }


  /**
   * Crea los campos del formulario con sus respectivas validaciones
   */
  private crearFormulario(): void {
    this.formulario = this.fb.group({
      fechaEmpalme: ['', Validators.compose([
        Validators.required
      ])],
      estadiaTemporal: [],
      tipoAtencion: ['', Validators.compose([
        Validators.required])],
      pisoDomiciliario: ['', Validators.compose([
        Validators.required])],
      pisoAmbulatorio: ['', Validators.compose([
        Validators.required])],
      programa: ['', Validators.compose([
        Validators.required])]
    });
  }

  /**
   * Convierte el campo celular en obligatorio u opcional
   * @param {boolean} obligatorio
   */
  private activarValidadores(activar: any): void {
    let controlesEmpalme = ['fechaEmpalme'];
    let controlesAdmision = ['tipoAtencion', 'pisoDomiciliario', 'pisoAmbulatorio', 'programa'];
    if (activar.checked) {
      this.cambiarValidadoresFormularioARequerido(controlesEmpalme);
      this.cambiarValidadoresFormularioAOpcional(controlesAdmision);
    } else {
      this.cambiarValidadoresFormularioARequerido(controlesAdmision);
      this.cambiarValidadoresFormularioAOpcional(controlesEmpalme);
    }
  }

  /**
   * Convierte el campo celular en obligatorio u opcional
   * @param {boolean} obligatorio
   */
  private activarValidadoresTipoAtencion(activar: any): void {
    let controlesEmpalme = ['pisoAmbulatorio'];
    let controlesAdmision = ['tipoAtencion', 'pisoDomiciliario', 'fechaEmpalme', 'programa'];
    if (activar.value) {
      this.cambiarValidadoresFormularioARequerido(controlesEmpalme);
      this.cambiarValidadoresFormularioAOpcional(controlesAdmision);
    } else {
      controlesEmpalme = ['pisoAmbulatorio', 'fechaEmpalme'];
      this.cambiarValidadoresFormularioARequerido(controlesAdmision);
      this.cambiarValidadoresFormularioAOpcional(controlesEmpalme);
    }
  }


  /**
   * Cambia los validadores del formulario
   * @param {string} nombrecontrol
   * @param {number} numeroDigitos
   */
  private cambiarValidadoresFormularioARequerido(controles: string[]): void {
    for (var i = 0; i < controles.length; i++) {
        this.formulario.get(controles[i]).clearValidators();
        this.formulario.get(controles[i]).setValidators(
          Validators.compose([
            Validators.required
          ])
        );
        this.formulario.get(controles[i]).updateValueAndValidity();
    }
  }

  /**
   * Abre el modal
   */
  private abrirModalCentroEstadia(): void {
    let dialogRef = this.dialog.open(ModalComponent, {
      width: '30%',
      disableClose: true,
      data: {  }
    });

    dialogRef.afterClosed().subscribe(result => {
      result ? this.cerrarRemision() : this.reiniciarCentroEstadia();
    });
  }

  /**
   * Reinicia el campo del formulario
   */
  private reiniciarCentroEstadia(): void {
    this.admisionViewModel.centroEstadiaTemporal = null;
    this.formulario.controls['estadiaTemporal'].setValue(false);
  }

  /**
   * Cierra la remision
   */
  private cerrarRemision(): void {
    this.formulario.controls['estadiaTemporal'].setValue(false);
  }

  /**
   * Cambia los validadores del formulario
   * @param {string} nombrecontrol
   * @param {number} numeroDigitos
   */
  private cambiarValidadoresFormularioAOpcional(controles: string[]): void {
      for (var i = 0; i < controles.length; i++) {
        this.formulario.get(controles[i]).clearValidators();
        this.formulario.get(controles[i]).setValidators(
          Validators.compose([])
        );
        this.formulario.get(controles[i]).updateValueAndValidity();
      }
  }

  /**
   * Inicializa variables del view model
   */
  private iniciarViewModel(): AdmisionViewModel {
    let fechaInicial: Date = new Date();
    fechaInicial.setDate(fechaInicial.getDate() + 1);
    let fechaFinal: Date = new Date();
    fechaFinal.setDate(fechaFinal.getDate() + 8);
    return new AdmisionViewModel(
      null,
      null,
      new Respuesta(null, '', null),
      new Respuesta(null, '', null),
      false,
      fechaInicial,
      fechaFinal,
      null,
       new Admision(
           '2/27/2018'
         , 'oscapara'
         , false
         , null
         , false
         , null
         , null
         , null
         , null
         , false
         , false
         , null
       ),
      null
    );
  }

  /**
   * Realiza llamado al servicio
   */
  private getPisos(): void {
    this.cargando.emit(true);
    this.admisionService.getPisos('1')
      .subscribe(
        response => {
          if (response) {
            this.admisionViewModel.respuestaPisos = response;

            if (response.success === false) {
              this.mensajesService.mostrarMensajeError(response.businessMessage);
            }
          }
        },
        error => {
          this.mensajesService.mostrarMensajeError(error.message);
        },
        () => {
          this.cargando.emit(false);
        }
      );
  }

  /**
   * Realiza llamado al servicio
   */
  private getProgramas(): void {
    this.cargando.emit(true);
    this.admisionService.getProgramas()
      .subscribe(
        response => {
          if (response) {
            this.admisionViewModel.respuestaProgramas = response;

            if (response.success === false) {
              this.mensajesService.mostrarMensajeError(response.businessMessage);
            }
          }
        },
        error => {
          this.mensajesService.mostrarMensajeError(error.message);
        },
        () => {
          this.cargando.emit(false);
        }
      );
  }

}
