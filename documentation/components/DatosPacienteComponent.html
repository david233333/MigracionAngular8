import { Component, OnInit, Output,
  EventEmitter, OnDestroy, Input,
  OnChanges, SimpleChange } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Subscription } from 'rxjs/Subscription';

import * as moment from 'moment';

import { DatosPacienteService } from '../../../domain/usecase/remision/datos-paciente.service';
import { DatosPacienteViewModel } from './datos-paciente.view-model';
import { Respuesta } from '../../../infraestructure/comun/models/respuesta.model';
import { MensajesService } from '../../../shared/services/mensajes.service';
import { TipoIdentificacion } from '../../../domain/model/comun/entity/tipo-identificacion.model';
import { MaestrosSec } from '../../../infraestructure/comun/models/maestro.model';
import { Paciente } from '../../../domain/model/remision/entity/paciente.model';

@Component({
  selector: 'sura-datos-paciente',
  templateUrl: './datos-paciente.component.html',
  styleUrls: ['./datos-paciente.component.scss']
})
export class DatosPacienteComponent implements OnInit, OnDestroy, OnChanges {
  @Input('tiposIdentificacion')
  private tiposIdentificacion: Respuesta<TipoIdentificacion[]>;

  @Output()
  private continuar: EventEmitter<boolean> = new EventEmitter<boolean>();

  @Output()
  private datosPaciente: EventEmitter<Paciente> = new EventEmitter<Paciente>();

  @Output()
  private edadPaciente: EventEmitter<number> = new EventEmitter<number>();

  private formulario: FormGroup;

  private datosPacienteViewModel: DatosPacienteViewModel = this.iniciarViewModel();

  private datosPacienteSubscripcion = new Subscription();
  private planSaludSubscripcion = new Subscription();
  private tiposPlanesSaludSubscripcion = new Subscription();

  constructor (private fb: FormBuilder,
               private datosPacienteService: DatosPacienteService,
               private mensajesService: MensajesService) {
    this.crearFormulario();
    this.suscribirFormulario();
  }

  ngOnInit() {
    this.getDatos();
  }

  ngOnDestroy() {
    this.datosPacienteSubscripcion.unsubscribe();
    this.planSaludSubscripcion.unsubscribe();
    this.tiposPlanesSaludSubscripcion.unsubscribe();
  }

  ngOnChanges(changes: { [propKey: string]: SimpleChange }) {
    if (changes && changes.tiposIdentificacion) {
      this.getTiposIdentificacion();
    }
  }

  /**
   * Obtiene los datos del paciente
   */
  private getDatos(): void {
    this.getTiposIdentificacion();
    this.getPlanesSalud();
    this.getTiposPlanesSalud();
  }

  /**
  * Suscribe los cambios del formulario                                                                                                                                                                                 rio
   */
  private suscribirFormulario(): void {
    this.formulario.valueChanges.subscribe(
      formulario => {
        this.continuar.emit(this.formulario.valid);
        this.mostrarCampos(formulario);
        this.limpiarDatosPaciente();
      }
    );

    this.formulario.get('planSalud').valueChanges.subscribe(
      valor => {
        this.cambiarValidezTipoPlanSalud(this.datosPacienteViewModel.mostrarTipoPlanSalud);
      }
    );
  }

  /**
   * Muestra u oculta los campos ARL y Particular
   */
  private mostrarCampos(formulario: any): void {
    this.datosPacienteViewModel.mostrarCodigoAutorizacion = formulario.planSalud === '7' || formulario.planSalud === '15';
    this.datosPacienteViewModel.mostrarTipoPlanSalud = formulario.planSalud === '3';
  }

  /**
   * Cambia los validadores del tipo de plan de salud
   */
  private cambiarValidezTipoPlanSalud(mostrarTipoPlanSalud: boolean): void {
    if (mostrarTipoPlanSalud === true) {
      this.formulario.get('tipoPlanSalud').setValidators(
        Validators.compose([Validators.required])
      );
    } else {
      this.formulario.get('tipoPlanSalud').clearValidators();
    }
    this.formulario.get('tipoPlanSalud').updateValueAndValidity();
  }

  /**
   * Inicializa variables del view model
   */
  private iniciarViewModel(): DatosPacienteViewModel {
    return new DatosPacienteViewModel(
      null,
      null,
      null,
      new Respuesta(null, '', null),
      new Respuesta(null, '', null),
      new Respuesta(null, '', null),
      new Respuesta(null, '', null),
      null,
      null
    );
  }

  /**
   * Limpia los datos del paciente al cambiar datos del formulario
   */
  private limpiarDatosPaciente(): void {
    this.datosPacienteViewModel.respuestaDatosPaciente = new Respuesta(null, '', null);
  }

  /**
   * Busca la informacion del paciente
   */
  private buscar(): void {
    let tipoDocumento: string = this.formulario.get('tipoDocumento').value;
    let documento: string = this.formulario.get('numeroDocumento').value;
    let planSalud: string = this.formulario.get('planSalud').value;

    this.getDatosPaciente(planSalud, tipoDocumento, documento);
  }

  /**
   * Realiza llamado al servicio
   * @param {string} planSalud
   * @param {string} tipoDocumento
   * @param {string} documento
   */
  private getDatosPaciente(planSalud: string, tipoDocumento: string, documento: string): void {
    this.datosPacienteSubscripcion =
      this.datosPacienteService.getDatosPaciente(planSalud, tipoDocumento, documento)
      .subscribe(
        response => {
          if (response) {
            this.datosPacienteViewModel.respuestaDatosPaciente = response;
            this.mostrarColumnas();

            response.datos ? this.datosPaciente.emit(response.datos) : this.datosPaciente.emit(null);

            this.validarEdadPaciente(response.datos);

            if (response.success === false) {
              this.mensajesService.mostrarMensajeError(response.businessMessage);
            }
          }
        },
        error => {
          this.mensajesService.mostrarMensajeError(error.message);
        },
        () => {
        }
      );
  }

  /**
   * Valida la edad del paciente
   * @param {Paciente} datos
   */
  private validarEdadPaciente(datos: Paciente): void {
    if (datos && datos.edad) {
      this.edadPaciente.emit(+datos.edad);
    }
  }

  /**
   * Obtiene los tipos de identificacion
   */
  private getTiposIdentificacion(): void {
    if (this.datosPacienteViewModel.respuestaTiposIdentificacion) {
      this.datosPacienteViewModel.respuestaTiposIdentificacion = this.tiposIdentificacion;
    }
  }

  /**
   * Realiza llamado al servicio
   */
  private getPlanesSalud(): void {
    this.planSaludSubscripcion =
      this.datosPacienteService.getPlanesSalud()
      .subscribe(
        response => {
          if (response) {
            this.datosPacienteViewModel.respuestaPlanesSalud = response;

            if (response.success === false) {
              this.mensajesService.mostrarMensajeError(response.businessMessage);
            }
          }
        },
        error => {
          this.mensajesService.mostrarMensajeError(error.message);
        },
        () => {

        }
      );
  }

  /**
   * Realiza llamado al servicio
   */
  private getTiposPlanesSalud(): void {
    this.tiposPlanesSaludSubscripcion =
      this.datosPacienteService.getTiposPlanesSalud()
        .subscribe(
          response => {
            if (response && response.datos) {
              if (response.datos[0].maestrosSec) {
                this.datosPacienteViewModel.respuestaTiposPlanesSalud =
                  new Respuesta<MaestrosSec[]>(
                    true,
                    null,
                    response.datos[0].maestrosSec
                  );
              }

              if (response.success === false) {
                this.mensajesService.mostrarMensajeError(response.businessMessage);
              }
            }
          },
          error => {
            this.mensajesService.mostrarMensajeError(error.message);
          },
          () => {

          }
        );
  }

  /**
   * Muestra u oculta las columnas
   */
  private mostrarColumnas(): void {
    this.datosPacienteViewModel.mostrarColumnaAsegurador = this.mostrarColumnasAsegurador();
    this.datosPacienteViewModel.mostrarColumnaAfiliacion = this.mostrarColumnasAfiliacion();
  }

  /**
   * Indica si se muestra o no la columna
   * @returns {boolean}
   */
  private mostrarColumnasAsegurador(): boolean {
    let datos = this.datosPacienteViewModel.respuestaDatosPaciente.datos;
    let mostrar: boolean = false;

    if (datos) {
      mostrar =
        datos.tipoAsegurador && datos.tipoAsegurador  !== '' ||
        datos.estadoSuspension && datos.estadoSuspension  !== '' ||
        datos.coberturaDomiciliaria && datos.coberturaDomiciliaria  !== null ||
        datos.fechaLimiteCobertura && datos.fechaLimiteCobertura !== '';
    }

    return mostrar;
  }

  /**
   * Indica si se muestra o no la columna
   * @returns {boolean}
   */
  private mostrarColumnasAfiliacion(): boolean {
    let datos = this.datosPacienteViewModel.respuestaDatosPaciente.datos;
    let mostrar: boolean = false;

    if (datos) {
      mostrar =
        datos.tipoAfiliacion && datos.tipoAfiliacion  !== '' ||
        datos.nivelIngreso && datos.nivelIngreso  !== '' ||
        datos.ipsBasicaAsignada && datos.ipsBasicaAsignada  !== '' ||
        datos.lugarAtencion && datos.lugarAtencion !== '';
    }

    return mostrar;
  }

  /**
   * Crea los campos del formulario con sus respectivas validaciones
   */
  private crearFormulario(): void {
    this.formulario = this.fb.group({
      tipoDocumento: ['', Validators.compose([
        Validators.required
      ])],
      numeroDocumento: ['', Validators.compose([
        Validators.required,
        Validators.maxLength(15),
        Validators.pattern('[0-9]+')
      ])],
      planSalud: ['', Validators.compose([
        Validators.required
      ])],
      codigoAutorizacion: ['', Validators.compose([
        Validators.maxLength(30)
      ])],
      tipoPlanSalud: ['', Validators.compose([])]
    });
  }
}
