import { Component, OnInit, Output, EventEmitter, AfterContentChecked, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Subscription } from 'rxjs/Subscription';

import { MatDialog } from '@angular/material';

import { ModalDireccionComponent } from './modal-direccion/modal-direccion.component';
import { ModalCondicionPacienteAceptaComponent } from './modal-condicion-paciente-acepta/modal-condicion-paciente-acepta.component';

import { Respuesta } from '../../../infraestructure/comun/models/respuesta.model';
import { DatosAtencionViewModel } from './datos-atencion.view-model';

import { DatosAtencionService } from '../../../domain/usecase/remision/datos-atencion.service';
import { MensajesService } from '../../../shared/services/mensajes.service';

@Component({
  selector: 'sura-datos-atencion',
  templateUrl: './datos-atencion.component.html',
  styleUrls: ['./datos-atencion.component.scss']
})
export class DatosAtencionComponent implements OnInit, AfterContentChecked, OnDestroy {
  @Output()
  private continuar: EventEmitter<boolean> = new EventEmitter<boolean>();

  @Output()
  private datosAtencion: EventEmitter<FormGroup> = new EventEmitter<FormGroup>();

  @Output()
  private ciudad: EventEmitter<string> = new EventEmitter<string>();

  private formulario: FormGroup;

  private datosAtencionViewModel: DatosAtencionViewModel = this.iniciarViewModel();

  private ciudadesSubscripcion: Subscription = new Subscription();
  private municipiosSubscripcion: Subscription = new Subscription();

  constructor(private fb: FormBuilder,
              private datosAtencionService: DatosAtencionService,
              private dialog: MatDialog,
              private mensajesService: MensajesService) {
    this.crearFormulario();
    this.suscribirFormulario();
  }

  ngOnInit() {
    this.getDatos();
  }

  ngAfterContentChecked() {
    this.desactivarDireccion();
  }

  ngOnDestroy() {
   this.ciudadesSubscripcion.unsubscribe();
   this.municipiosSubscripcion.unsubscribe();
  }

  /**
   * Inicializa variables del view model
   */
  private iniciarViewModel(): DatosAtencionViewModel {
    return new DatosAtencionViewModel(
      null,
      '',
      '',
      null,
      null,
      new Respuesta(null, '', null),
      new Respuesta(null, '', null)
    );
  }

  /**
   * Obtiene los datos de atencion
   */
  private getDatos(): void {
    this.datosAtencionViewModel.placeholderTelefono =
      this.datosAtencionViewModel.mensajes.campos.telefonoPacienteRequerido;
    this.datosAtencionViewModel.placeholderCelular =
      this.datosAtencionViewModel.mensajes.campos.celularPaciente;
    this.getCiudades();
  }

  /**
   * Emite la ciudad seleccionada
   * @param evento
   */
  private emitirCiudad(evento): void {
    if (evento && evento.value) {
      this.ciudad.emit(evento.value);
    }
  }

  /**
  * Suscribe los cambios del formulario                                                                                                                                                                                 rio
  */
  private suscribirFormulario(): void {
    this.formulario.valueChanges.subscribe(
      formulario => {
        this.continuar.emit(this.formulario.valid);
        this.datosAtencion.emit(formulario);
      }
    );
  }

  /**
   * Evento cuanto cambia la ciudad seleccionada
   * @param evento
   */
  private buscarMunicipios(evento) {
    if (evento && evento.value) {
      this.getMunicipios(evento.value);
    } else {
      this.datosAtencionViewModel.respuestaMunicipios = new Respuesta(null, '', null);
    }
  }

  /**
   * Realiza llamado al servicio
   */
  private getCiudades(): void {
    this.ciudadesSubscripcion =
      this.datosAtencionService.getCiudades()
        .subscribe(
          response => {
            if (response) {
              this.datosAtencionViewModel.respuestaCiudades = response;

              if (response.success === false) {
                this.mensajesService.mostrarMensajeError(response.businessMessage);
              }
            }
          },
          error => {
            this.mensajesService.mostrarMensajeError(error.message);
          },
          () => {

          }
        );
  }

  /**
   * Obtiene los municipios por ciudad
   * @param {string} idCiudad
   */
  private getMunicipios(idCiudad: string): void {
    if (idCiudad) {
      this.municipiosSubscripcion = this.datosAtencionService.getMunicipios(idCiudad)
        .subscribe(
          response => {
            if (response) {
              this.datosAtencionViewModel.respuestaMunicipios = response;
            }
          },
          error => {
            this.mensajesService.mostrarMensajeError(error.message);
          },
          () => {

          }
        );
    }
  }

  /**
   * Desactiva el campo direccion
   */
  private desactivarDireccion(): void {
    let controlDireccion = this.formulario.get('direccion');
    controlDireccion.disable();
  }

  /**
   * Abre el modal
   */
  private abrirModalCondicionPacienteAcepta(): void {
      let dialogRef = this.dialog.open(ModalCondicionPacienteAceptaComponent, {
        width: '30%',
        disableClose: true,
        data: {  }
      });

    dialogRef.afterClosed().subscribe(result => {
      result ? this.cerrarRemision() : this.reiniciarCondicionPacienteAcepta();
    });
  }

  /**
   * Abre el modal
   */
  private abrirModalDireccion(): void {
    let dialogRef = this.dialog.open(ModalDireccionComponent, {
      width: '99%',
      disableClose: true,
      data: {  }
    });

    dialogRef.afterClosed().subscribe(result => {
      result ? this.cerrarRemision() : this.reiniciarCondicionPacienteAcepta();
    });
  }

  /**
   * Reinicia el campo del formulario
   */
  private reiniciarCondicionPacienteAcepta(): void {
    this.formulario.get('condicionPacienteAcepta').reset();
    this.datosAtencionViewModel.condicionPacienteAcepta = null;
  }

  /**
   * Cierra la remision
   */
  private cerrarRemision(): void {

  }

  /**
   * Metodo que se ejecuta al cambiar alguno de los telefonos
   */
  private cambiarTelefono(): void {
    this.validarTelefonos();
  }

  /**
   * Cambia el campo del formulario
   * @param event
   */
  private cambiarCondicionPacienteAcepta(event): void {
    let valor = event.value;
    this.formulario.get('condicionPacienteAcepta').setValue(valor);
    if (!valor) {
      this.abrirModalCondicionPacienteAcepta();
    }
  }

  /**
   * Cambia el campo del formulario
   * @param event
   */
  private cambiarCondicionServicios(event): void {
    let valor = event.value;
    this.formulario.get('condicionServicios').setValue(valor);
  }

  /**
   * Asigna validaciones de los telefonos
   */
  private validarTelefonos(): void {
    if (this.formulario.get('telefonoPaciente').value === '' &&
      this.formulario.get('celularPaciente').value === '') {
      this.telefonoObligatorioCelularOpcional();
    }

    if (this.formulario.get('telefonoPaciente').value !== '' &&
      this.formulario.get('celularPaciente').value === '') {
      this.telefonoObligatorioCelularOpcional();
    }

    if (this.formulario.get('telefonoPaciente').value !== '' &&
      this.formulario.get('celularPaciente').value !== '') {
      this.telefonoObligatorioCelularOpcional();
    }

    if (this.formulario.get('telefonoPaciente').value === '' &&
      this.formulario.get('celularPaciente').value !== '') {
      this.telefonoOpcionalCelularObligatorio();
    }
  }

  /**
   * Convierte el campo telefono obligatorio y el campo celular opcional
   */
  private telefonoObligatorioCelularOpcional(): void {
    this.convertirTelefonoObligatorio(true);
    this.convertirCelularObligatorio(false);
  }

  /**
   * Convierte el campo telefono opcional y el campo celular obligatorio
   */
  private telefonoOpcionalCelularObligatorio(): void {
    this.convertirTelefonoObligatorio(false);
    this.convertirCelularObligatorio(true);
  }

  /**
   * Convierte el campo telefono en obligatorio u opcional
   * @param {boolean} obligatorio
   */
    private convertirTelefonoObligatorio(obligatorio: boolean): void {
      if (obligatorio) {
      this.datosAtencionViewModel.placeholderTelefono = this.datosAtencionViewModel.mensajes.campos.telefonoPacienteRequerido;
      this.cambiarValidadoresFormularioARequerido('telefonoPaciente', 7);
    } else {
      this.datosAtencionViewModel.placeholderTelefono = this.datosAtencionViewModel.mensajes.campos.telefonoPaciente;
      this.cambiarValidadoresFormularioAOpcional('telefonoPaciente', 7);
    }
    this.formulario.get('telefonoPaciente').updateValueAndValidity();
  }

  /**
   * Convierte el campo celular en obligatorio u opcional
   * @param {boolean} obligatorio
   */
  private convertirCelularObligatorio(obligatorio: boolean): void {
    if (obligatorio) {
      this.datosAtencionViewModel.placeholderCelular = this.datosAtencionViewModel.mensajes.campos.celularPacienteRequerido;
      this.cambiarValidadoresFormularioARequerido('celularPaciente', 10);
    } else {
      this.datosAtencionViewModel.placeholderCelular = this.datosAtencionViewModel.mensajes.campos.celularPaciente;
      this.cambiarValidadoresFormularioAOpcional('celularPaciente', 10);
    }
    this.formulario.get('celularPaciente').updateValueAndValidity();
  }

  /**
   * Cambia los validadores del formulario
   * @param {string} nombrecontrol
   * @param {number} numeroDigitos
   */
  private cambiarValidadoresFormularioARequerido(nombrecontrol: string, numeroDigitos: number): void {
    this.formulario.get(nombrecontrol).clearValidators();
    this.formulario.get(nombrecontrol).setValidators(
      Validators.compose([
        Validators.required,
        Validators.maxLength(numeroDigitos),
        Validators.pattern('[0-9]+')
      ])
    );
    this.formulario.get(nombrecontrol).updateValueAndValidity();
  }

  /**
   * Cambia los validadores del formulario
   * @param {string} nombrecontrol
   * @param {number} numeroDigitos
   */
  private cambiarValidadoresFormularioAOpcional(nombrecontrol: string, numeroDigitos: number): void {
    this.formulario.get(nombrecontrol).clearValidators();
    this.formulario.get(nombrecontrol).setValidators(
      Validators.compose([
        Validators.maxLength(numeroDigitos),
        Validators.pattern('[0-9]+')
      ])
    );
    this.formulario.get(nombrecontrol).updateValueAndValidity();
  }

  /**
   * Retorna valor de campo del formulario indicado
   * @param {string} nombreControl
   * @returns {string}
   */
  private obtenerValorFormulario(nombreControl: string): string {
    return this.formulario.get(nombreControl).value;
  }

  /**
   * Crea los campos del formulario con sus respectivas validaciones
   */
  private crearFormulario(): void {
    this.formulario = this.fb.group({
      condicionPacienteAcepta: [ null, Validators.compose([
        Validators.required
      ])],
      condicionServicios: [ null, Validators.compose([
        Validators.required
      ])],
      ciudad: [ '', Validators.compose([
        Validators.required
      ])],
      municipio: [ '', Validators.compose([
        Validators.required
      ])],
      barrio: [ '', Validators.compose([
        Validators.required
      ])],
      direccion: [ '', Validators.compose([
        Validators.required
      ])],
      nombreCuidador: [ '', Validators.compose([
        Validators.maxLength(50)
      ])],
      nombreResponsable: [ '', Validators.compose([
        Validators.maxLength(50)
      ])],
      telefonoPaciente: [ '', Validators.compose([
        Validators.required,
        Validators.maxLength(7),
        Validators.pattern('[0-9]+')
      ])],
      celularPaciente: [ '', Validators.compose([
        Validators.maxLength(10),
        Validators.pattern('[0-9]+')
      ])],

    });
  }

}
